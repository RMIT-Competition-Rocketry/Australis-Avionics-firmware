cmake_minimum_required(VERSION 3.15)

project(AV2-demo_supervisor LANGUAGES C ASM)

# ╔══════════════════════════════════════════════════════════╗
# ║                          BUILD                           ║
# ╚══════════════════════════════════════════════════════════╝

build_australis_variant(
  NAME ${PROJECT_NAME}
  EXTRA_MODULES 
  VARIANT_SOURCES
    "src/main.c"
  VARIANT_INCLUDES
    "inc"
)

  LINKER:-Map=${MAP_DIR}/${PROJECT_NAME}.map
)
