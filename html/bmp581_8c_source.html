27BMP581_t BMP581_init(
28 BMP581_t *baro,
29 GPIO_TypeDef *port,
30 unsigned long cs,
31 float tempSensitivity,
32 float pressSensitivity
33) {
34 SPI_init(&baro->base, SPI1, MODE8, port, cs);
35 baro->tempSensitivity = tempSensitivity;
36 baro->pressSensitivity = pressSensitivity;
37 baro->update = BMP581_update;
38 baro->readTemp = BMP581_readTemp;
39 baro->readRawTemp = BMP581_readRawTemp;
40 baro->processRawTemp = BMP581_processRawTemp;
41 baro->readPress = BMP581_readPress;
42 baro->readRawPress = BMP581_readRawPress;
43 baro->processRawPress = BMP581_processRawPress;
44
45 // Initial dummy read
46 BMP581_readRegister(baro, 0x01);
47
48 // Soft reset device
49 BMP581_writeRegister(baro, BMP581_CMD, 0xB6);
50
51 while (BMP581_readRegister(baro, BMP581_CHIP_ID) == 0x00); // Check chip ID
52 while (BMP581_readRegister(baro, BMP581_INT_STATUS) != 0x10); // Wait for POR complete
53 while (!(BMP581_readRegister(baro, BMP581_STATUS) & BMP581_STATUS_NVM_RDY)); // Check device status NVM ready
54 while ((BMP581_readRegister(baro, BMP581_STATUS) & BMP581_STATUS_NVM_ERR)); // Check device status NVM err
55
56 volatile uint8_t counter = 0;
57
58 BMP581_writeRegister(baro, BMP581_ODR_CFG, BMP581_ODR_CFG_DEEP_DIS); // Disable deep sleep
59 for (uint32_t i = 0; i < 0x1FFFF; i++) {
60 counter++;
61 } // Wait for at least t_standby
62 BMP581_writeRegister(baro, BMP581_ODR_CFG, BMP581_ODR_CFG_DEEP_DIS | BMP581_ODR_CFG_PWR_CONTINUOUS); // Set continuous sample
63
64 uint8_t OSRCFG = BMP581_readRegister(baro, BMP581_OSR_CFG);
65 BMP581_writeRegister(baro, BMP581_OSR_CFG, (BMP581_OSR_CFG_RESERVED & OSRCFG) | BMP581_OSR_CFG_PRESS_EN | BMP581_OSR_CFG_OSR_P_16);
66
67 // Set ground pressure reading on init
68 for (uint32_t i = 0; i < 0x1FFFF; i++) {
69 counter++;
70 } // Wait for at least t_reconf
71 baro->readPress(baro, &baro->groundPress); // Read current pressure
72
73 return *baro;
74}

178void BMP581_writeRegister(BMP581_t *baro, uint8_t address, uint8_t data) {
179 SPI spi = baro->base;
180
181 spi.port->ODR &= ~spi.cs;
182
183 // Send read command and address
184 uint8_t payload = address & 0x7F; // Load payload with address and read command
185 spi.transmit(&spi, payload); // Transmit payload
186 spi.transmit(&spi, data); // Transmit dummy data and read response data
187
188 spi.port->ODR |= spi.cs;
189}

191uint8_t BMP581_readRegister(BMP581_t *baro, uint8_t address) {
192 uint8_t response = 0;
193 SPI spi = baro->base;
194
195 spi.port->ODR &= ~spi.cs;
196
197 // Send read command and address
198 uint8_t payload = address | 0x80; // Load payload with address and read command
199 response = spi.transmit(&spi, payload); // Transmit payload
200 response = spi.transmit(&spi, 0xFF); // Transmit dummy data and read response data
201
202 spi.port->ODR |= spi.cs;
203
204 return response;
205}

207void BMP581_readRegisters(BMP581_t *baro, uint8_t address, uint8_t count, uint8_t *out) {
208 SPI spi = baro->base;
209
210 spi.port->ODR &= ~spi.cs;
211
212 // Send read command and address
213 uint8_t payload = address | 0x80; // Load payload with address and read command
214 spi.transmit(&spi, payload); // Transmit payload
215
216 // Auto increment read through registers
217 for (uint8_t i = 0; i < count; i++) {
218 out[i] = spi.transmit(&spi, 0xFF);
219 }
220
221 spi.port->ODR |= spi.cs;
222}
