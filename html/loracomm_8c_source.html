35void vLoRaTransmit(void *argument) {
36 const TickType_t blockTime = portMAX_DELAY;
37 uint8_t rxData[LORA_MSG_LENGTH];
38
39 SX1272_t *lora = DeviceList_getDeviceHandle(DEVICE_LORA).device;
40 GPIOpin_t rfToggle = GPIOpin_init(GPIOE, GPIO_PIN2, NULL);
41
42 for (;;) {
43 // Wait for SX1272 to be ready for transmission
44 EventBits_t uxBits = xEventGroupWaitBits(xMsgReadyGroup, GROUP_MESSAGE_READY_LORA, pdFALSE, pdFALSE, blockTime);
45 if ((uxBits & GROUP_MESSAGE_READY_LORA)) {
46 // Wait to receive message in buffer
47 size_t xReceivedBytes = xMessageBufferReceive(
48 xLoRaTxBuff,
49 (void *)rxData,
50 sizeof(rxData),
51 blockTime
52 );
53 // Transmit if message is available
54 if (xReceivedBytes) {
55 rfToggle.reset(&rfToggle); // Toggle RF front-end for transmit
56 lora->transmit(lora, rxData);
57 xEventGroupClearBits(xMsgReadyGroup, GROUP_MESSAGE_READY_LORA);
58 }
59 }
60 }

71void vLoRaSample(void *argument) {
72 const TickType_t xFrequency = pdMS_TO_TICKS(250);
73 const TickType_t blockTime = pdMS_TO_TICKS(125);
74
75 A3G4250D_t *gyro = DeviceList_getDeviceHandle(DEVICE_GYRO).device;
76 KX134_1211_t *lAccel = DeviceList_getDeviceHandle(DEVICE_ACCEL_LOW).device;
77 KX134_1211_t *hAccel = DeviceList_getDeviceHandle(DEVICE_ACCEL_HIGH).device;
78
79 enum State *flightState = StateHandle_getHandle("FlightState").state;
80 float *altitude = StateHandle_getHandle("Altitude").state;
81 float *velocity = StateHandle_getHandle("Velocity").state;
82
83 for (;;) {
84 // Block until 250ms interval
85 TickType_t xLastWakeTime = xTaskGetTickCount();
86 vTaskDelayUntil(&xLastWakeTime, xFrequency);
87
88 uint8_t systemStatus = xEventGroupGetBits(xSystemStatusGroup);
89
90 // Create AVData packet with current data
91 SX1272_Packet avData = SX1272_AVData(
92 LORA_HEADER_AV_DATA,
93 *flightState | systemStatus,
94 lAccel->rawAccelData,
95 hAccel->rawAccelData,
96 KX134_1211_DATA_TOTAL,
97 gyro->rawGyroData,
98 A3G4250D_DATA_TOTAL,
99 *altitude,
100 *velocity
101 );
102
103 // Add packet to queue
104 xMessageBufferSend(xLoRaTxBuff, &avData, LORA_MSG_LENGTH, blockTime);
105 }
106}
