44int Shell_init(Shell *shell) {
45 shell->usb = *(UART_t *)DeviceList_getDeviceHandle(DEVICE_UART_USB).device;
46 shell->help = Shell_help;
47 shell->run = Shell_run;
48 shell->runTask = Shell_runTask;
49 shell->clear = Shell_clear;
50
51 // Return error code if shell vector exceeds max allowable programs
52 if (((&__shell_vector_end - &__shell_vector_start) / sizeof(uint32_t)) > SHELL_MAX_PROGRAMS)
53 return 1;
54
55 // Register programs in vector to shell
56 int idx = 0;
57 for (uint32_t *i = (uint32_t *)&__shell_vector_start; i < (uint32_t *)&__shell_vector_end; i++) {
58 // Dereference function pointer
59 ShellProgramHandle_t (*registerShellProgram)() = (ShellProgramHandle_t(*)())(*i);
60
61 // Register program handle to shell
62 shell->programHandles[idx] = registerShellProgram();
63 idx++;
64 }
65
66 return 0;
67}

103void Shell_run(Shell *shell, uint8_t *programName) {
104 char *token = strtok((char *)programName, " ");
105 char *flags = strchr(token, '\0') + 1;
106
107 // Iterate through shell vector and execute function from handle
108 // with matching name (if any).
109 for (int i = 0; i < SHELL_MAX_PROGRAMS; i++) {
110 if (!strcmp(shell->programHandles[i].name, programName)) {
111 shell->programHandles[i].exec(shell, flags);
112 return; // Early exit if program is found
113 }
114 }
115
116 // Print help string if no matching command is found
117 shell->usb.print(&shell->usb, (char *)programName);
118 shell->usb.print(&shell->usb, ": command not recognized. Run `help` for a list of available commands\n\r");
119}

163 return true;
