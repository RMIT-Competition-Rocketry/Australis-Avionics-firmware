29bool initDevices() {
30 static DeviceHandle_t deviceList[DEVICE_MAX_KEYS];
31 DeviceList_init(deviceList);
32
33 // ==========================================================================
34 // HIGH RANGE ACCELEROMETER
35 //
36 // Provides low resolution (high scale) inertial data for 3-axis acceleration.
37 // This is used for data logging as well as state estimation when
38 // the rocket's upward velocity is greater than the maximum high
39 // resolution scale.
40 static KX134_1211_t hAccel;
41 KX134_1211_init(
42 &hAccel,
43 ACCEL_PORT_2, // GPIO port connecting accelerometer 2 CS pin
44 ACCEL_CS_2, // Position of accelerometer 2 CS pin in GPIO prt
45 ACCEL_SCALE_HIGH, // Set to high resolution for larger G forces
46 ACCEL_AXES_2, // Accelerometer 2 mounting axes
47 ACCEL_SIGN_2 // +/- for mounting axes
48 );
49 deviceList[DEVICE_ACCEL_HIGH].deviceName = "HAccel";
50 deviceList[DEVICE_ACCEL_HIGH].device = &hAccel;
51
52 // ==========================================================================
53 // LOW RANGE ACCELEROMETER
54 //
55 // Provides high resolution (low scale) inertial data for 3-axis acceleration.
56 // This is used for data logging as well as state estimation when
57 // the rocket's upward velocity is within range of the high resolution.
58 static KX134_1211_t lAccel;
59 KX134_1211_init(
60 &lAccel,
61 ACCEL_PORT_1, // GPIO port connecting accelerometer 1 CS pin
62 ACCEL_CS_1, // Position of accelerometer 1 CS pin in GPIO prt
63 ACCEL_SCALE_LOW, // Set to low resolution for smaller G forces
64 ACCEL_AXES_1, // Accelerometer 1 mounting axes
65 ACCEL_SIGN_1 // +/- for mounting axes
66 );
67 deviceList[DEVICE_ACCEL_LOW].deviceName = "LAccel";
68 deviceList[DEVICE_ACCEL_LOW].device = &lAccel;
69
70 // ==========================================================================
71 // BAROMETER
72 //
73 // Measures temperature compensated atmospheric pressure. Data from the
74 // barometer is used in altitude calculation, as well as providing one
75 // metric for detecting apogee (decreasing pressure readings).
76 static BMP581_t baro;
77 BMP581_init(
78 &baro,
79 BARO_PORT, // GPIO port connecting barometer CS pin
80 BARO_CS, // Position of barometer CS pin in GPIO prt
81 BMP581_TEMP_SENSITIVITY, // Set temperature measurement sensitivity
82 BMP581_PRESS_SENSITIVITY // Set pressure measurement sensitivity
83 );
84 deviceList[DEVICE_BARO].deviceName = "Baro";
85 deviceList[DEVICE_BARO].device = &baro;
86
87 // ==========================================================================
88 // GYROSCOPE
89 //
90 // Measures inertial data for 3-axis rotations. This data is used in
91 // calculations for attitude quaternion to determine rotation during flight

93 static A3G4250D_t gyro;
94 A3G4250D_init(
95 &gyro,
96 GYRO_PORT, // GPIO port connecting gyroscope CS pin
97 GYRO_CS, // Position of gyroscope CS pin in GPIO prt
98 A3G4250D_SENSITIVITY, // Set measurement sensitivity
99 GYRO_AXES, // Gyroscope mounting axes
100 GYRO_SIGN // +/- for mounting axes
101 );
102 deviceList[DEVICE_GYRO].deviceName = "Gyro";
103 deviceList[DEVICE_GYRO].device = &gyro;
104
105 // ==========================================================================
106 // FLASH
107 //
108 // Flash storage for measured and calculated data during flight.

110 W25Q128_init(
111 &flash,
112 FLASH_PORT, // GPIO port connecting flash CS pin

117 deviceList[DEVICE_FLASH].deviceName = "Flash";

120 // ==========================================================================
121 // USB UART
122 //

146 GPS_INTERFACE, // Memory mapped address of UART interface for GPS
147 GPS_PORT, // GPIO port connecting UART data pins
148 GPS_PINS, // Position of data pins in GPIO prt

151 deviceList[DEVICE_GPS].deviceName = "GPS";

154 // ==========================================================================
155 // LORA
156 //
157 // LoRa transceiver for external wireless communicatons. Can be configured to
158 // either receive or transmit data.
159 static SX1272_t lora;
160 SX1272_init(
161 &lora,
162 LORA_PORT, // GPIO port connecting LoRa CS pin
163 LORA_CS, // Position of LoRa CS pin in GPIO prt
164 BW500, // Set LoRa bandwidth to 500KHz
165 SF9, // Spreading factor 9
166 CR5 // Coding rate 4/5
167 );
168 deviceList[DEVICE_LORA].deviceName = "LoRa";
169 deviceList[DEVICE_LORA].device = &lora;
170
171 // ==========================================================================
172 // !! THIS DEVICE HANDLE WILL BE MODIFIED AT RUN-TIME !!
173 // DEVICE_ACCEL represents the current accelerometer selected by the system.
174 //
175 // The intent is that the high frequency data task will determine which of the
176 // two connected accelerometers (configured for high and low resolution) will be
177 // used for data acquisition. Other tasks will should interact with the current
178 // accelerometer without directly modifying it.
179 deviceList[DEVICE_ACCEL] = deviceList[DEVICE_ACCEL_LOW];
180
181 return true;
182}

189StateHandle_t StateHandle_getHandle(char *name) {
190 StateHandle_t *handleRef = StateHandle_getHandleRef(name);
191 return (handleRef == NULL) ? (StateHandle_t){"NULL", NULL} : *handleRef;
192}
193
194StateHandle_t *StateHandle_getHandleRef(char *name) {
195 // Iterate through all handles in State vector
196 for (uint8_t *i = (uint8_t *)&__state_vector_start; i < (uint8_t *)&__state_vector_end; i += sizeof(StateHandle_t)) {
197 StateHandle_t *handle = (StateHandle_t *)i;
198 // Return State handle if names match
199 if (!strcmp(handle->name, name)) {
200 handle->ref = handle;
201 return handle;
202 }
203 }
204 // Return NULL pointer if no matching State is found
205 return NULL;
206}
