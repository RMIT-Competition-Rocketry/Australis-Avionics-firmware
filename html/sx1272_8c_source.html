28SX1272_t SX1272_init(
29 SX1272_t *lora,
30 GPIO_TypeDef *port,
31 unsigned long cs,
32 Bandwidth bw,
33 SpreadingFactor sf,
34 CodingRate cr
35) {
36 SPI_init(&lora->base, SPI3, MODE16, port, cs);
37 lora->transmit = SX1272_transmit;
38
39 _SX1272_setMode(lora, SLEEP); // Set mode to sleep
40
41 // Set interrupt pin
42 SX1272_writeRegister(lora, RegDioMapping1, 0x40);
43
44 /* clang-format off */
45 SX1272_writeRegister(lora, LORA_REG_OP_MODE,
46 0x01 << LORA_REG_OP_MODE_LONG_RANGE_Pos // Enable LoRa
47 );
48
49 SX1272_writeRegister(lora, LORA_REG_MODEM_CONFIG1,
50 bw << LORA_REG_MODEM_CONFIG1_BW_Pos // Set bandwidth
51 | cr << LORA_REG_MODEM_CONFIG1_CR_Pos // Set coding rate
52 | 0x00 << LORA_REG_MODEM_CONFIG1_CRC_Pos // Enable CRC
53 );
54 /* clang-format on */
55
57 SX1272_writeRegister(lora, LORA_REG_MODEM_CONFIG2, 0x94);
58
59 // Set payload length
60 SX1272_writeRegister(lora, LORA_REG_PAYLOAD_LENGTH, LORA_MSG_LENGTH);
61 SX1272_writeRegister(lora, LORA_REG_MAX_PAYLOAD_LENGTH, LORA_MSG_LENGTH);
62
63 _SX1272_setMode(lora, STDBY); // Set mode to standby
64
65 return *lora;
66}

109SX1272_Packet SX1272_AVData(
110 uint8_t id,
111 uint8_t currentState,
112 uint8_t *lAccelData,
113 uint8_t *hAccelData,
114 uint8_t lenAccel,
115 uint8_t *gyroData,
116 uint8_t lenGyro,
117 float altitude,
118 float velocity
119) {
120 SX1272_Packet msg;
121
122 // Convert altitude float to byte array
123 union {
124 float f;
125 uint8_t b[4];
126 } a;
127 a.f = altitude;
128
129 // Convert velocity float to byte array
130 union {
131 float f;
132 uint8_t b[4];
133 } v;
134 v.f = velocity;
135
136 int idx = 0;
137 // Append to struct data array
138 msg.id = id;
139 msg.data[idx++] = currentState;
140 memcpy(&msg.data[idx], lAccelData, lenAccel);
141 memcpy(&msg.data[idx += lenAccel], hAccelData, lenAccel);
142 memcpy(&msg.data[idx += lenAccel], gyroData, lenGyro);
143 memcpy(&msg.data[idx += lenGyro], a.b, sizeof(float));
144 memcpy(&msg.data[idx += sizeof(float)], v.b, sizeof(float));
145
146 return msg;
147}

194void SX1272_transmit(SX1272_t *lora, uint8_t *pointerdata) {
195 SX1272_writeRegister(lora, LORA_REG_IRQ_FLAGS, 0x08); // clears the status flags
196 SX1272_writeRegister(lora, LORA_REG_FIFO_ADDR_PTR, 0x80); // set pointer adddress to TX
197
198 // Load data into transmit FIFO
199 for (int i = 0; i < 32; i++) {
200 SX1272_writeRegister(lora, LORA_REG_FIFO, pointerdata[i]);
201 }
202
203 // Set device to transmit
204 _SX1272_setMode(lora, TX);
205
206 // Clear the status flags
207 SX1272_writeRegister(lora, LORA_REG_IRQ_FLAGS, 0x08);
208}

212void SX1272_writeRegister(SX1272_t *lora, uint8_t address, uint8_t data) {
213 SPI spi = lora->base;
214
215 uint16_t payload = (address << 0x08) | (1 << 0x0F); // Load payload with address and write command
216 payload |= data; // Append data to payload
217 spi.port->ODR &= ~spi.cs; // Lower chip select
218 spi.transmit(&spi, payload); // Send payload over SPI
219 spi.port->ODR |= spi.cs; // Raise chip select
220}
221
222uint8_t SX1272_readRegister(SX1272_t *lora, uint8_t address) {
223 SPI spi = lora->base;
224 uint16_t response;
225
226 uint16_t payload = (address << 0x08); // Load payload with address and read command
227 spi.port->ODR &= ~spi.cs; // Lower chip select
228 response = spi.transmit(&spi, payload); // Receive payload over SPI
229 spi.port->ODR |= spi.cs; // Raise chip select
230 return (uint8_t)response;
231}
