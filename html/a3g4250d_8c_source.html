26A3G4250D_t A3G4250D_init(
27 A3G4250D_t *gyro,
28 GPIO_TypeDef *port,
29 unsigned long cs,
30 float sensitivity,
31 const uint8_t *axes,
32 const int8_t *sign
33) {
34 SPI_init(&gyro->base, SPI1, MODE8, port, cs);
35 gyro->sensitivity = sensitivity;
36 gyro->update = A3G4250D_update;
37 gyro->readGyro = A3G4250D_readGyro;
38 gyro->readRawBytes = A3G4250D_readRawBytes;
39 gyro->processRawBytes = A3G4250D_processRawBytes;
40 memcpy(gyro->axes, axes, A3G4250D_DATA_COUNT);
41 memcpy(gyro->sign, sign, A3G4250D_DATA_COUNT);
42
43 const uint32_t superDelay = 0xFFFF;
44 volatile uint8_t counter = 0;
45
46 // Wait for the spefified period - need to wait for 2ms here.
47 for (uint32_t i = 0; i < superDelay; i++) {
48 counter++;
49 }
50
51 A3G4250D_writeRegister(gyro, A3G4250D_CTRL_REG1, A3G4250D_CTRL_REG1_ODR_800Hz | A3G4250D_CTRL_REG1_AXIS_ENABLE | A3G4250D_CTRL_REG1_PD_ENABLE);
52
53 return *gyro;
54}

96void A3G4250D_processRawBytes(A3G4250D_t *gyro, uint8_t *bytes, float *out) {
97 out[0] = gyro->sign[0] * gyro->sensitivity * (int16_t)(((uint16_t)bytes[0] << 8) | bytes[1]); // gyro X
98 out[1] = gyro->sign[1] * gyro->sensitivity * (int16_t)(((uint16_t)bytes[2] << 8) | bytes[3]); // gyro Y
99 out[2] = gyro->sign[2] * gyro->sensitivity * (int16_t)(((uint16_t)bytes[4] << 8) | bytes[5]); // gyro Z
100}

111void A3G4250D_readRawBytes(A3G4250D_t *gyro, uint8_t *out) {
112 #define INDEX_AXES(index, byte) 2 * gyro->axes[index] + byte
113 out[INDEX_AXES(0, 0)] = A3G4250D_readRegister(gyro, A3G4250D_OUT_X_H); // gyro X high
114 out[INDEX_AXES(0, 1)] = A3G4250D_readRegister(gyro, A3G4250D_OUT_X_L); // gyro X low
115 out[INDEX_AXES(1, 0)] = A3G4250D_readRegister(gyro, A3G4250D_OUT_Y_H); // gyro Y high
116 out[INDEX_AXES(1, 1)] = A3G4250D_readRegister(gyro, A3G4250D_OUT_Y_L); // gyro Y low
117 out[INDEX_AXES(2, 0)] = A3G4250D_readRegister(gyro, A3G4250D_OUT_Z_H); // gyro Z high
118 out[INDEX_AXES(2, 1)] = A3G4250D_readRegister(gyro, A3G4250D_OUT_Z_L); // gyro Z low
119#undef INDEX_AXES
120}

124void A3G4250D_writeRegister(A3G4250D_t *gyro, uint8_t address, uint8_t data) {
125 SPI spi = gyro->base;
126
127 spi.port->ODR &= ~spi.cs;
128
129 // Send read command and address
130 uint8_t payload = address & 0x7F; // Load payload with address and read command
131 spi.transmit(&spi, payload); // Transmit payload
132 spi.transmit(&spi, data); // Transmit dummy data and read response data
133
134 spi.port->ODR |= spi.cs;
135}
136
137uint8_t A3G4250D_readRegister(A3G4250D_t *gyro, uint8_t address) {
138 uint8_t response = 0;
139 SPI spi = gyro->base;
140
141 spi.port->ODR &= ~spi.cs;
142
143 // Send read command and address
144 uint8_t payload = address | 0x80; // Load payload with address and read command
145 response = spi.transmit(&spi, payload); // Transmit payload
146 response = spi.transmit(&spi, 0xFF); // Transmit dummy data and read response data
147
148 spi.port->ODR |= spi.cs;
149
150 return response;
151}
